\documentclass[aspectratio=169]{beamer}

\usetheme{metropolis}
\usecolortheme{default}
\metroset{block=fill}   % give blocks a filled background so they read as boxes
\setbeamercolor{background canvas}{bg=white}   % force a pure white background

% metropolis has no \framesubtitle template, so subtitles vanish. Redefine the
% frametitle template so the subtitle sits INSIDE the dark title bar, under the
% title (rather than spilling onto the white content area).
\setbeamercolor{framesubtitle}{fg=white}
\setbeamerfont{framesubtitle}{size=\small, series=\mdseries}
\makeatletter
\defbeamertemplate{frametitle}{metropolis-subtitle}{%
  \nointerlineskip%
  \begin{beamercolorbox}[%
      wd=\paperwidth,%
      sep=0pt,%
      leftskip=\metropolis@frametitle@padding,%
      rightskip=\metropolis@frametitle@padding,%
    ]{frametitle}%
  \metropolis@frametitlestrut@start%
  \insertframetitle%
  \ifx\insertframesubtitle\empty\else%
    \nolinebreak\\[0.25em]%
    {\usebeamerfont{framesubtitle}\usebeamercolor[fg]{framesubtitle}\insertframesubtitle}%
  \fi%
  \nolinebreak%
  \metropolis@frametitlestrut@end%
  \end{beamercolorbox}%
}
\makeatother
\setbeamertemplate{frametitle}[metropolis-subtitle]

% the title, which uses ~2.2ex of padding).

%   Talk:    make            -> main.pdf         clean slides, no notes

% -----------------------------------------------

  \fi
\fi
