      \begin{enumerate}

        \item train $q_\theta(X) \approx p_0(Y \mid X)$ on data from $D_n$
        \item predict $\mathbb{P}(Y = \hat{y}_{\text{test}} \mid X = x_{\text{test}}) = q_\theta(\hat{y}_{\text{test}} \mid x_{\text{test}})$
      \end{enumerate}

\begin{frame}{Background}
  \framesubtitle{Bayesian Framework}

  For every $n$, the statistical model gives a tuple $ (\mathcal{D}_n \cup (Y, X), p). $
  
  \vspace{0.25em}

  We want to approximate $p(Y \mid X)$ by
  \[
    \pi(y \mid x, D_n)
    \overset{\text{def}}{\equiv}
    \mathbb{P}\big(Y = y \mid X = x;\, \mathcal{D}_n = D_n\big)
  \]

  \onslide<2->{
  With
  \[
    q_\theta(X, \mathcal{D}_n) \approx \pi(Y \mid X, \mathcal{D}_n)
  \]
  }

  \onslide<3->{
  \vspace{-1.0em}
  And make predictions
  \[
    \mathbb{P}(Y = \hat{y}_{\text{test}} \mid X = x_{\text{test}}, \mathcal{D}_n = D_n) = q_\theta(\hat{y}_{\text{test}} \mid x_{\text{test}}, D_n)
  \]
  }
  \vspace{1.0em}

  \onslide<4->{
  \vspace{-2.5em}
  \begin{alertblock}{}
    \centering $\pi$ is called the \textbf{Posterior Predictive Distribution}
  \end{alertblock}
  }
\end{frame}

\begin{frame}{Background}
  \framesubtitle{Meta Level}

  \small
  \begin{block}{Prior $\mathbb{P}_\Phi$}
    \begin{center}
      Define a hypothesis space $\Phi$ of DGPs where each hypothesis $\varphi \in \Phi$ is s.t. \ ($y \mid x) \sim \varphi$.
    \end{center}

    \vspace{-1.5em}
    \begin{center}
      The distribution $\mathbb{P}_\Phi$ is called \textbf{prior} and expresses our beliefs about $\varphi$ before seeing any data.
    \end{center}
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \centering
      {\footnotesize $ $}\\[0.2em]
      \metapanel{$\mathbb{P}_{\Phi(\cdot)}(\varphi)$}{0.3}{0.3}{0.3}{0.3}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      {\footnotesize $ $}\\[0.2em]
      \metapanel{$\mathbb{P}_{\Phi(\cdot \mid \cdot)}(\varphi \mid D_n)$}{2.0}{0.05}{0.6}{0.05}
    \end{column}
  \end{columns}
\end{frame}


\begin{frame}{Background}
  \framesubtitle{Posterior Predictive Distribution}

  \onslide<1->{
  \[
    \pi(y \mid {\alt<3>{\textcolor{red}{x}}{x},} \text{ }D_n)
    = \int \only<2>{\color{blue!90!olive}}\varphi(y \mid x)\only<2>{\color{black}}\, \only<2>{\color{red!70!olive}}d\mathbb{P}_{\Phi(\cdot \mid \cdot)}\only<2>{\color{black}}(\varphi \mid \alt<4>{}{\alt<3>{\textcolor{red}{x}}{x}, }D_n) \tag{1}
  \]
  }

  \onslide<2->{
  \begin{align*}
    = \; & \only<2>{\color{red!70!olive}}\mathbb{P}\big(\varphi = \varphi_1 = \text{\scalebox{0.45}{\hypLinA}} \mid \alt<4>{}{\alt<3>{\textcolor{red}{x}}{x}, }D_n\big)\only<2>{\color{black}} \cdot \only<2>{\color{blue!90!olive}}\varphi_1(y \mid x)\only<2>{\color{black}} \\
      + \; & \only<2>{\color{red!70!olive}}\mathbb{P}\big(\varphi = \varphi_2 = \text{\scalebox{0.45}{\hypLinB}} \mid \alt<4>{}{\alt<3>{\textcolor{red}{x}}{x}, }D_n\big)\only<2>{\color{black}} \cdot \only<2>{\color{blue!90!olive}}\varphi_2(y \mid x)\only<2>{\color{black}} \; + \; \dots
  \end{align*}
  }

  \onslide<3->{
  \vspace{0.5em}
  \begin{alertblock}{Assumption}
    \centering $\mathbb{P}(\varphi \mid x, D_n) = \mathbb{P}(\varphi \mid D_n)$ \quad i.e. $x \equiv x_{\text{test}}$ is not informative about hypothesis $\varphi$.
  \end{alertblock}
  }
\end{frame}

\begin{frame}{Background}
  \framesubtitle{Posterior Predictive Distribution}

  Dropping $x$ from the conditioning on $\varphi$:
  \[
    \pi(y \mid x, D_n)
    = \int \varphi(y \mid x)\, d\mathbb{P}_{\Phi(\cdot \mid \cdot)}(\varphi \mid D_n)
    = \int p(y \mid x, \varphi) \cdot \underbrace{p(\varphi \mid D_n)}_{\text{not available}} \, d\varphi
  \]

  \onslide<2->{
  \vspace{0.5em}
  By Bayes' rule, $p(\varphi \mid D_n) \propto p(D_n \mid \varphi)\, p(\varphi)$, so
  \[
    \pi(y \mid x, D_n) \;\propto\; \int
    \color{olive}{\underbrace{\color{black}p(y \mid x, \varphi)}_{\substack{\text{prediction based} \\ \text{on hypothesis } \varphi}}}
    \color{black}\cdot
    \color{teal}\underbrace{\color{black}p(D_n \mid \varphi)}_{\substack{\text{how likely is} \\ D_n \text{ under } \varphi}}
    \color{black}\cdot
    \color{blue}\underbrace{\color{black}p(\varphi)}_{\substack{\text{prior belief} \\ \text{about } \varphi}}
    \, \color{black}d\varphi
  \]
  }
\end{frame}

\begin{frame}{Background}
  \framesubtitle{Prior-Data Fitted Networks (PFNs)}

  \small
  \setlength{\abovedisplayskip}{3pt}\setlength{\belowdisplayskip}{2pt}
  \setlength{\abovedisplayshortskip}{2pt}\setlength{\belowdisplayshortskip}{2pt}
  \textbf{Prior-Data Fitted Network} is a numerical approximation of the entire \emph{family} of PPDs $\{\pi(\cdot \mid \cdot,\, \mathcal{D}_n)\}_{n}$.

  \vspace{-0.2em}
  \onslide<1->{
  \begin{block}{Theorem 2.2. \cite{naglerStatisticalFoundationsPriorData2023}}
    Let $\mathcal{Q}$ be the set of \emph{all} conditional probability functions. Then the PPD satisfies
    \[
      \pi \;=\; \arg\max_{q \in \mathcal{Q}}\; \mathbb{E}_{{\color{blue}\varphi \sim \mathbb{P}_{\Phi}},\, {\color{teal}\mathcal{D}_n \cup (Y, X) \sim (\mathcal{D}_{n+1} \mid \varphi)}}\big[\log {\color{olive}q(Y \mid X, \mathcal{D}_n)}\big].
    \]
  \end{block}
  }

  \onslide<2->{
  \begin{center}
    $\Downarrow$\enspace {\footnotesize restrict to a parametric model $q_\theta$}
  \end{center}
  }

  \vspace{-1.5em}
  \begin{gather*}
    \onslide<2->{\theta^{*}       = \arg\max_{\theta}\; \mathbb{E}_{N \sim \mathbb{P}_N,\, {\color{blue}\varphi \sim \mathbb{P}_{\Phi}},\, {\color{teal}\mathcal{D}_n \cup (Y, X) \sim (\mathcal{D}_{n+1} \mid \varphi)}}\big[\log {\color{olive}q_{\theta}(Y \mid X, \mathcal{D}_N)}\big]} \\[0.25em]
    \onslide<3->{\Downarrow\ \ \text{\footnotesize approximate in practice by Monte-Carlo}} \\[0.25em]
    \onslide<3->{\widehat{\theta} = \arg\max_{\theta}\; \sum_{j=1}^{m} \log q_{\theta}\big(Y_j \mid X_j, D^{(j)}\big)}
  \end{gather*}

  \note{The expectation in Theorem~2.2 follows the hierarchical data-generating mechanism: first draw the true model (data-generating process) $\varphi \sim \Phi$; then generate the training sample $D_n$ and the test pair $(Y, X)$ from that selected $\varphi$.}
\end{frame}
